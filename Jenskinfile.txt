pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo 'Building the application...'
                // Use Maven as the build automation tool
                sh 'mvn clean package'
            }
        }
        stage('Unit and Integration Tests') {
            steps {
                echo 'Running unit and integration tests...'
                // Use JUnit or TestNG for unit tests, and Selenium for integration tests
                sh 'mvn test'
            }
            post {
                always {
                    echo 'Sending notification email for Tests stage...'
                    mail to: 'alissatran1108@gmail.com',
                         subject: "Pipeline Status: Tests Stage - ${currentBuild.fullDisplayName}",
                         body: "Pipeline result for Tests stage: ${currentBuild.result}\nLogs: ${currentBuild.url}console"
                }
            }
        }
        stage('Code Analysis') {
            steps {
                echo 'Analyzing code quality...'
                // Use SonarQube for code analysis
                sh 'mvn sonar:sonar'
            }
        }
        stage('Security Scan') {
            steps {
                echo 'Performing security scan...'
                // Use OWASP Dependency-Check for security scanning
                sh 'dependency-check --project project --scan ./ --format XML --out dependency-check-report.xml'
            }
            post {
                always {
                    echo 'Sending notification email for Security Scan stage...'
                    mail to: 'alissatran1108@gmail.com',
                         subject: "Pipeline Status: Security Scan Stage - ${currentBuild.fullDisplayName}",
                         body: "Pipeline result for Security Scan stage: ${currentBuild.result}\nLogs: ${currentBuild.url}console"
                }
            }
        }
        stage('Deploy to Staging') {
            steps {
                echo 'Deploying to staging environment...'
                // Use AWS CLI or Ansible for deployment
                sh 'aws deploy create-deployment --application-name myApp --deployment-group-name staging'
            }
        }
        stage('Integration Tests on Staging') {
            steps {
                echo 'Running integration tests on staging...'
                // Use Selenium or Postman for integration testing in staging
                sh 'mvn verify'
            }
        }
        stage('Deploy to Production') {
            steps {
                echo 'Deploying to production environment...'
                // Use AWS CLI or Ansible for deployment
                sh 'aws deploy create-deployment --application-name myApp --deployment-group-name production'
            }
        }
    }
}
